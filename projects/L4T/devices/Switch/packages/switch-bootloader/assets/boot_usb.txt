# mkimage -A arm -T script -O linux -d boot_usb.txt boot.scr
mmc dev 1
load mmc 1:1 0x83000000 lakka/KERNEL
load mmc 1:1 0x8d000000 lakka/boot/tegra210-nintendo-switch.dtb
setenv bootargs 'console=ttyGS0 fbcon=rotate:3 boot=/dev/mmcblk0p1 ro rootwait loglevel=5 ssh tty keep_bootcon no_console_suspend pmc_reboot2payload.reboot_action=bootloader pmc_reboot2payload.default_payload=reboot_payload.bin pmc_reboot2payload.hekate_config_id=LAKKA'
usb reset
booti 0x83000000 - 0x8d000000
